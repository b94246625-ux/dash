rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {



match /user_tokens/{userId} {
  allow read, write: if request.auth != null
                     && request.auth.uid == userId;
}


    match /trip_requests/{requestId} {
// 🚗 Driver: update live location
allow update: if request.auth != null
  && resource.data.driverId == request.auth.uid
  && resource.data.status == 'accepted'
  && request.resource.data.diff(resource.data)
       .affectedKeys()
       .hasOnly([
         'driverLocation',
         'driverLat',
         'driverLng',
         'driverHeading',
         'lastLocationUpdate'
       ]);
      // 👑 Admin
      allow read, write: if request.auth != null
        && request.auth.token.admin == true;

      // 👨‍👩‍👧 Parent
      allow read: if request.auth != null
        && resource.data.parentId == request.auth.uid;

      allow create: if request.auth != null
        && request.resource.data.parentId == request.auth.uid;

      // 🚗 Driver reads
      allow read: if request.auth != null
        && resource.data.status == 'pending';

      allow read: if request.auth != null
        && resource.data.driverId == request.auth.uid
        && resource.data.status == 'accepted';

      // 🚗 Accept trip
      allow update: if request.auth != null
        && resource.data.status == 'pending'
        && request.resource.data.status == 'accepted'
        && request.resource.data.driverId == request.auth.uid
        && request.resource.data.diff(resource.data)
             .affectedKeys()
             .hasOnly(['status', 'driverId', 'acceptedAt']);

      // 🚗 Update ETA
      allow update: if request.auth != null
        && resource.data.driverId == request.auth.uid
        && resource.data.status == 'accepted'
        && request.resource.data.diff(resource.data)
             .affectedKeys()
             .hasOnly(['etaMinutes']);

      // 🚗 Update trip stage
      allow update: if request.auth != null
        && resource.data.driverId == request.auth.uid
        && resource.data.status == 'accepted'
        && request.resource.data.status == 'accepted'
        && request.resource.data.driverId == resource.data.driverId
        && request.resource.data.parentId == resource.data.parentId
        && request.resource.data.childId == resource.data.childId
        && request.resource.data.diff(resource.data)
             .affectedKeys()
             .hasOnly(['tripStage', 'pickedAt', 'droppedAt']);
    }

    match /parents/{parentId} {
      allow create, read, update: if request.auth != null
        && request.auth.uid == parentId;

      allow delete: if false;

      match /children/{childId} {
        allow read, write: if request.auth != null
          && request.auth.uid == parentId;

        match /schedules/{scheduleId} {
          allow read, write: if request.auth != null
            && request.auth.uid == parentId;
        }
      }

      match /saved_locations/{locationId} {
        allow read, write: if request.auth != null
          && request.auth.uid == parentId;
      }
    }

    match /drivers/{driverId} {
      allow read: if request.auth != null;

      allow read, write: if request.auth != null
        && request.auth.token.admin == true;

      allow create, read: if request.auth != null
        && request.auth.uid == driverId;

      allow update: if request.auth != null
        && request.auth.uid == driverId
        && isDriverUpdateAllowed();

      allow delete: if false;

      function isDriverUpdateAllowed() {
        let changed = request.resource.data.diff(resource.data).affectedKeys();
        return !changed.hasAny([
          'isApproved',
          'kyc.status',
          'kyc.license.verified',
          'kyc.greyCard.verified',
          'kyc.reviewedAt',
          'kyc.reviewNotes'
        ]);
      }
    }

    match /user_tokens/{userId} {
      allow read, write: if request.auth != null
        && request.auth.uid == userId;
    }

  }
}
